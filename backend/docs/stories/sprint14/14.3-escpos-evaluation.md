### 3.2 EscPosAdapter 抽象草案

```kotlin
// backend/src/main/kotlin/com/btsheng/erp/print/protocol/EscPosAdapter.kt (V1.3.10 backlog 草案)
interface PrintProtocolAdapter {
    fun build(payload: PrintPayload): ByteArray
}

class EscPosAdapter : PrintProtocolAdapter {
    // ESC/POS 字节流生成
    // ESC @ 初始化 + ESC ! 字号 + ESC a 对齐 + GS V 切纸 + GS ( k 二维码
    override fun build(payload: PrintPayload): ByteArray {
        val bytes = ByteArrayOutputStream()
        bytes.write(0x1B); bytes.write('@'.code)  // ESC @ 初始化
        // ... 字号 + 对齐 + 文本 + 二维码 + 切纸
        return bytes.toByteArray()
    }
}
```
