## 3. 端点契约（消费 Sprint 10.3 既有端点 · 不新增）

```yaml
# backend/spec/openapi.yaml (Sprint 10.3 已 ship · Sprint 13.4 消费)
paths:
  /api/v1/workflow/events/stats:
    get:
      tags: [Workflow]
      summary: 工作流事件统计报表
      operationId: getWorkflowEventStats
      security:
        - bearerAuth: []
      parameters:
        - in: query
          name: workflow_code
          schema: { type: string, enum: [PO_APPROVAL, QUOTE_APPROVAL, OUTSOURCE_APPROVAL] }
          description: 工作流代码 · 不传则全部
        - in: query
          name: approver_role
          schema: { type: string, enum: [PROCUREMENT_MANAGER, SALES_MANAGER, GM, ADMIN] }
          description: 审批人角色 · 不传则全部
        - in: query
          name: start_date
          schema: { type: string, format: date }
        - in: query
          name: end_date
          schema: { type: string, format: date }
      responses:
        '200':
          description: 统计结果
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/WorkflowEventStatsResponse'
        '403':
          description: 非 GM + ADMIN（40301 GM_OR_ADMIN_REQUIRED）

components:
  schemas:
    WorkflowEventStatsResponse:
      type: object
      properties:
        totalCount:
          type: integer
          description: 总审批事件数
        byEventType:
          type: object
          additionalProperties:
            type: integer
          description: 按事件类型分布（SUBMITTED/APPROVED/REJECTED/REASSIGNED）
        byApproverRole:
          type: object
          additionalProperties:
            type: integer
          description: 按审批人角色工作量（PROCUREMENT_MANAGER/SALES_MANAGER/GM/ADMIN）
        period:
          type: object
          properties:
            start: { type: string, format: date }
            end: { type: string, format: date }
        version: { type: string, description: "V1.3.9.0" }
```

> **Sprint 13.4 消费**：组件 mount 时调此端点 · 4 图渲染 byEventType（饼图）+ byApproverRole（柱状图）+ period 趋势（折线）+ totalCount（数字卡）。

### 4.2 WorkflowEventPanel.vue

```vue
<!-- web-impl/src/views/v138/components/WorkflowEventPanel.vue (新增) -->
<template>
  <div class="workflow-event-panel">
    <!-- 过滤器 -->
    <el-form :inline="true" :model="filters">
      <el-form-item label="工作流">
        <el-select v-model="filters.workflowCode" placeholder="全部" clearable>
          <el-option label="PO 审批" value="PO_APPROVAL" />
          <el-option label="报价审批" value="QUOTE_APPROVAL" />
          <el-option label="委外审批" value="OUTSOURCE_APPROVAL" />
        </el-select>
      </el-form-item>
      <el-form-item label="审批角色">
        <el-select v-model="filters.approverRole" placeholder="全部" clearable>
          <el-option label="采购主管" value="PROCUREMENT_MANAGER" />
          <el-option label="销售主管" value="SALES_MANAGER" />
          <el-option label="总经理" value="GM" />
          <el-option label="管理员" value="ADMIN" />
        </el-select>
      </el-form-item>
      <el-form-item label="开始日期">
        <el-date-picker v-model="filters.startDate" type="date" />
      </el-form-item>
      <el-form-item label="结束日期">
        <el-date-picker v-model="filters.endDate" type="date" />
      </el-form-item>
      <el-form-item>
        <el-button type="primary" @click="loadStats">查询</el-button>
      </el-form-item>
    </el-form>

    <!-- 加载骨架 -->
    <el-skeleton v-if="loading" :rows="6" animated />

    <!-- 4 图渲染 -->
    <el-row v-else :gutter="20">
      <el-col :span="6">
        <!-- 1. 总数数字卡 -->
        <el-card>
          <div class="metric-card">
            <div class="metric-label">总审批事件</div>
            <div class="metric-value">{{ stats?.totalCount ?? 0 }}</div>
          </div>
        </el-card>
      </el-col>
      <el-col :span="9">
        <!-- 2. 按事件类型饼图 -->
        <el-card title="按事件类型">
          <v-chart :option="eventTypeChartOption" autoresize />
        </el-card>
      </el-col>
      <el-col :span="9">
        <!-- 3. 按审批角色柱状图 -->
        <el-card title="按审批角色工作量">
          <v-chart :option="approverRoleChartOption" autoresize />
        </el-card>
      </el-col>
      <el-col :span="24">
        <!-- 4. 时间序列折线图 -->
        <el-card title="审批趋势">
          <v-chart :option="periodChartOption" autoresize />
        </el-card>
      </el-col>
    </el-row>
  </div>
</template>

<script setup lang="ts">
import { ref, computed, onMounted } from 'vue'
import VChart from 'vue-echarts'
import { use } from 'echarts/core'
import { CanvasRenderer } from 'echarts/renderers'
import { PieChart, BarChart, LineChart } from 'echarts/charts'
import { TitleComponent, TooltipComponent, LegendComponent, GridComponent } from 'echarts/components'
import { workflowApi } from '@/api/generated/services/WorkflowService'
import type { WorkflowEventStatsResponse } from '@/api/generated/models/WorkflowEventStatsResponse'

use([CanvasRenderer, PieChart, BarChart, LineChart, TitleComponent, TooltipComponent, LegendComponent, GridComponent])

const filters = ref({
  workflowCode: undefined as string | undefined,
  approverRole: undefined as string | undefined,
  startDate: undefined as Date | undefined,
  endDate: undefined as Date | undefined
})

const stats = ref<WorkflowEventStatsResponse | null>(null)
const loading = ref(false)

async function loadStats() {
  loading.value = true
  try {
    const response = await workflowApi.getWorkflowEventStats({
      workflowCode: filters.value.workflowCode,
      approverRole: filters.value.approverRole,
      startDate: filters.value.startDate?.toISOString().slice(0, 10),
      endDate: filters.value.endDate?.toISOString().slice(0, 10)
    })
    stats.value = response.data
  } finally {
    loading.value = false
  }
}

onMounted(loadStats)

// ECharts option 计算属性
const eventTypeChartOption = computed(() => ({
  tooltip: { trigger: 'item' },
  series: [{
    type: 'pie',
    radius: '60%',
    data: Object.entries(stats.value?.byEventType ?? {}).map(([name, value]) => ({ name, value }))
  }]
}))

const approverRoleChartOption = computed(() => ({
  tooltip: {},
  xAxis: { type: 'category', data: Object.keys(stats.value?.byApproverRole ?? {}) },
  yAxis: { type: 'value' },
  series: [{
    type: 'bar',
    data: Object.values(stats.value?.byApproverRole ?? {})
  }]
}))

const periodChartOption = computed(() => ({
  tooltip: { trigger: 'axis' },
  xAxis: { type: 'category', data: Object.keys(stats.value?.period ?? {}) },
  yAxis: { type: 'value' },
  series: [{ type: 'line', data: Object.values(stats.value?.period ?? {}), smooth: true }]
}))
</script>
```
