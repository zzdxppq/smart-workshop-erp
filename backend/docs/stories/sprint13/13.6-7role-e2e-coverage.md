// android-impl/src/androidTest/kotlin/com/btsheng/erp/e2e/RoleBasedE2ETest.kt (新增)
@RunWith(AndroidJUnit4::class)
@LargeTest
class RoleBasedE2ETest {
    @get:Rule
    val activityRule = ActivityScenarioRule(MainActivity::class.java)
    
    @Test
    fun testEngineerFullAccess() {
        // 1. 登录 ENGINEER 账号
        onView(withId(R.id.username)).perform(typeText("engineer_1001"))
        onView(withId(R.id.password)).perform(typeText("Test@123"))
        onView(withId(R.id.loginButton)).perform(click())
        
        // 2. 等待登录完成 → 进入主菜单
        onView(withId(R.id.mainMenu)).check(matches(isDisplayed()))
        
        // 3. 扫码 → 工单详情（GD-260614-001）
        onView(withId(R.id.scanButton)).perform(click())
        // ... 模拟扫码
        
        // 4. 图纸 Tab → 预览 → 打印 → 下载
        onView(withId(R.id.drawingTab)).perform(click())
        onView(withId(R.id.previewButton)).perform(click())
        // 验证 PDF 流渲染
        onView(withId(R.id.printButton)).perform(click())
        // 验证打印弹窗
        onView(withId(R.id.downloadButton)).perform(click())
        // 验证下载成功 Toast
    }
    
    @Test
    fun testFinanceNoAccess() {
        // 1. 登录 FINANCE 账号

        // 2. 主菜单 → 图纸菜单不可见
