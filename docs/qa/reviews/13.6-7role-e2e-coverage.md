## 三、CI 集成

### GitLab CI 配置

```yaml
# android-impl/.gitlab-ci.yml (新增 13.6 E2E)
e2e_7role_android:
  stage: e2e
  image: openjdk:17-jdk
  services:
    - name: erp-backend:latest
      alias: backend
  before_script:
    - cd android-impl
    - chmod +x gradlew
  script:
    # Test orchestrator 启用（自动清理 app 状态）
    - ./gradlew connectedAndroidTest \
        -Pandroid.testInstrumentationRunnerArguments.class=com.btsheng.erp.e2e.RoleBasedE2ETest \
        -Pandroid.testInstrumentationRunnerArguments.notAnnotation=android.test.suitebuilder.annotation.LargeTest
    # 4 worker parallel（device 池：4 AVD 实例）
    - ./gradlew connectedAndroidTest --max-workers=4
  artifacts:
    when: always
    paths:
      - android-impl/app/build/reports/androidTests/
      - android-impl/app/build/outputs/androidTest-results/
    expire_in: 30 days
  only:
    - merge_requests
    - main
  parallel: 4
  # 沙箱限制：本沙箱无法直接执行，委托 DevOps 张良
  when: manual
  allow_failure: false
```
