### 2.2 数据流

```ts
// useWorkflowStats.ts（核心 composable）
export function useWorkflowStats() {
  const stats = ref<WorkflowEventStatsResponse | null>(null)
  const loading = ref(false)
  const error = ref<Error | null>(null)

  async function fetchStats(params: {
    workflowCode?: string
    approverRole?: string
    startDate?: string
    endDate?: string
  }) {
    loading.value = true
    try {
      const { data } = await workflowApi.getWorkflowEventStats(params)
      stats.value = data
    } catch (e) {
      error.value = e
    } finally {
      loading.value = false
    }
  }

  return { stats, loading, error, fetchStats }
}

// WorkflowStats.vue
onMounted(() => {
  // 默认 7 天范围
  const end = new Date()
  const start = new Date(Date.now() - 7 * 24 * 60 * 60 * 1000)
  fetchStats({
    startDate: start.toISOString().slice(0, 10),
    endDate: end.toISOString().slice(0, 10)
  })
})
```

**数据流链路**：mounted → `useWorkflowStats.fetchStats()` → `workflowApi.getWorkflowEventStats()` → 10.3 端点 → 返回 `WorkflowEventStatsResponse` → 4 子组件 computed option 渲染。

## 6. IMPL 注意事项（3 条）

### ⚠️ 6.1 mounted 时调 10.3（不要在 setup 阶段）

```ts
// ❌ 错误：setup 阶段调用（SSR 风险）
const { stats, fetchStats } = useWorkflowStats()
fetchStats({ startDate: '...', endDate: '...' })

// ✅ 正确：onMounted 调用（避免 SSR hydration mismatch）
const { stats, fetchStats } = useWorkflowStats()
onMounted(() => {
  const end = new Date()
  const start = new Date(Date.now() - 7 * 24 * 60 * 60 * 1000)
  fetchStats({
    startDate: start.toISOString().slice(0, 10),
    endDate: end.toISOString().slice(0, 10)
  })
})
```

**原因**：setup 阶段在 SSR 服务端执行 · `new Date()` 服务端/客户端不一致 · 触发 hydration mismatch warning。`onMounted` 仅客户端执行。

### ⚠️ 6.2 ECharts option 用 shallowRef 包裹（避免深度响应式开销）

```ts
// ❌ 错误：ref 包裹（深度响应式 · ECharts option 含大量嵌套对象）
const eventTypeChartOption = ref({ tooltip: {...}, series: [{...}] })

// ✅ 正确：shallowRef 包裹（浅响应式 · 手动 triggerRef）
const eventTypeChartOption = shallowRef({})
watch(stats, (newStats) => {
  eventTypeChartOption.value = {
    tooltip: { trigger: 'item' },
    series: [{
      type: 'pie',
      radius: '60%',
      data: Object.entries(newStats?.byEventType ?? {}).map(([name, value]) => ({ name, value }))
    }]
  }
}, { deep: false })
```

**原因**：ECharts option 是大型嵌套对象 · `ref` 触发深度 Proxy 包装 · 性能开销大。`shallowRef` 仅包装 `.value` 引用 · 内部对象保持原始引用。

### ⚠️ 6.3 onUnmounted 销毁 ECharts 实例（避免内存泄漏）

```ts
// ✅ 正确：onUnmounted 销毁
import { onUnmounted, getCurrentInstance } from 'vue'

const chartRefs = ref<HTMLDivElement[]>([])

onUnmounted(() => {
  chartRefs.value.forEach(el => {
    const instance = echarts.getInstanceByDom(el)
    if (instance) instance.dispose()
  })
})
```

**原因**：ECharts 实例持有 Canvas/DOM 引用 · Vue 组件销毁后若不显式 `dispose()` · 内存泄漏 · 多次切换 Tab 后内存膨胀。
